import Foundation


public func solution(_ A : inout [Int]) -> Int {

    // write your code in Swift 4.2.1 (Linux)

    

    A = A.sorted()

    

    var iterator = 0

    

    for i in 0..<A.count - 1 {

        if (A[i] + 1) != A[i + 1] {

            iterator = (A[i] + 1)

            break

        }

    }

    

    return iterator

}
