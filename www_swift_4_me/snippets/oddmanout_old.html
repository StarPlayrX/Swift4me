import Foundation


public func solution(_ A : inout [Int]) -> Int {

    // write your code in Swift 4.2.1 (Linux)

    

    //let newArray = A.sorted()

    

    for i in A {

        

        let find = A.filter( { $0 == i } )

        

        if find.count == 1 {

            return i

        }

    }

    

    return 0

  

}
